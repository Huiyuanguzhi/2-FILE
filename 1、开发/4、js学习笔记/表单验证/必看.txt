	function execute(){
		var fu=document.getElementById("fu");
		var zi=document.getElementsByName("zi");
		if(fu.checked==true){
			for(var i=0;i<zi.length;i++){
				var poo=zi[i];
				poo.checked=true;
			}
		}else if(fu.checked==false){
			for(var i=0;zi.length;i++){
				var poo=zi[i];
				poo.checked=false;
			}
			
		}
	}

				count++;
			}
		}
		if(count>0){
			fu.checked=true;
		}else{
			fu.checked=false;
		}
	}
